    ```typescript
    export enum StepStatus {
      COMPLETED = "completed",
      CURRENT = "current",
      LOCKED = "locked",
    }

    export interface StepDefinition {
      number: number;        // 0-indexed step number
      name: string;          // Display name
      description: string;   // Short description
      icon: string;          // Emoji or icon identifier
    }

    export interface WizardState {
      currentStep: number;
      completedSteps: number[];
      getStepStatus: (stepNumber: number) => StepStatus;
      goToStep: (stepNumber: number) => void;
      completeCurrentStep: () => void;
      canNavigateToStep: (stepNumber: number) => boolean;
    }

    export const STEPS: StepDefinition[] = [
      { number: 0, name: "Connect Wallet", description: "Connect your EVM wallet to GOAT Testnet3", icon: "wallet" },
      { number: 1, name: "Agent Identity", description: "Register your ERC-8004 agent identity", icon: "identity" },
      { number: 2, name: "Payment Setup", description: "Configure x402 payment credentials", icon: "payment" },
      { number: 3, name: "Telegram", description: "Wire your Telegram bot connection", icon: "telegram" },
      { number: 4, name: "Demo Transaction", description: "Execute your first Hello GOAT transaction", icon: "rocket" },
    ];
    ```

    **Step 2: Create wizard state hook**
    Create `hooks/use-wizard.ts` with `"use client"`:

    Local state management for Phase 1 (no Convex wiring yet -- Plan 03 adds persistence).
    Use `useState` for `currentStep` (init 0) and `completedSteps` (init []).

    Implement:
    - `getStepStatus(stepNumber)`: returns COMPLETED if in completedSteps, CURRENT if === currentStep, LOCKED otherwise
    - `goToStep(stepNumber)`: only allows navigation to completed steps (for read-only view) or current step
    - `completeCurrentStep()`: adds currentStep to completedSteps, advances currentStep by 1 (cap at STEPS.length - 1)
    - `canNavigateToStep(stepNumber)`: true if step is completed or is the current step
    - Return type matches WizardState interface
